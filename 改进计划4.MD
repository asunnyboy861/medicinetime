```swift
// MARK: - Timeline Provider
struct MedicationProvider: TimelineProvider {
    func placeholder(in context: Context) -> MedicationEntry {
        MedicationEntry(
            date: Date(),
            totalCount: 12,
            expiringCount: 3,
            lowStockCount: 2,
            expiringMedications: [
                WidgetMedication(name: "Tylenol", daysUntilExpiry: 5, isExpired: false),
                WidgetMedication(name: "Advil", daysUntilExpiry: 12, isExpired: false),
                WidgetMedication(name: "Vitamin D", daysUntilExpiry: -2, isExpired: true)
            ]
        )
    }
    
    func getSnapshot(in context: Context, completion: @escaping (MedicationEntry) -> ()) {
        let entry = loadMedicationData()
        completion(entry)
    }
    
    func getTimeline(in context: Context, completion: @escaping (Timeline<Entry>) -> ()) {
        var entries: [MedicationEntry] = []
        let currentDate = Date()
        
        // Refresh every hour
        for hourOffset in 0..<6 {
            let entryDate = Calendar.current.date(byAdding: .hour, value: hourOffset, to: currentDate)!
            let entry = loadMedicationData(for: entryDate)
            entries.append(entry)
        }
        
        let timeline = Timeline(entries: entries, policy: .atEnd)
        completion(timeline)
    }
    
    private func loadMedicationData(for date: Date = Date()) -> MedicationEntry {
        // Load data from shared Core Data or UserDefaults
        let sharedDefaults = UserDefaults(suiteName: "group.com.yourcompany.medicinetime")
        
        let totalCount = sharedDefaults?.integer(forKey: "widget_totalCount") ?? 0
        let expiringCount = sharedDefaults?.integer(forKey: "widget_expiringCount") ?? 0
        let lowStockCount = sharedDefaults?.integer(forKey: "widget_lowStockCount") ?? 0
        
        // Load expiring medications from shared storage
        let medications = loadExpiringMedications()
        
        return MedicationEntry(
            date: date,
            totalCount: totalCount,
            expiringCount: expiringCount,
            lowStockCount: lowStockCount,
            expiringMedications: medications
        )
    }
    
    private func loadExpiringMedications() -> [WidgetMedication] {
        // Load from shared Core Data or decode from UserDefaults
        guard let sharedDefaults = UserDefaults(suiteName: "group.com.yourcompany.medicinetime"),
              let data = sharedDefaults.data(forKey: "widget_expiringMedications"),
              let medications = try? JSONDecoder().decode([WidgetMedication].self, from: data) else {
            return []
        }
        return medications
    }
}

// MARK: - Entry Model
struct MedicationEntry: TimelineEntry {
    let date: Date
    let totalCount: Int
    let expiringCount: Int
    let lowStockCount: Int
    let expiringMedications: [WidgetMedication]
}

// MARK: - Widget Medication Model
struct WidgetMedication: Codable, Identifiable {
    let id = UUID()
    let name: String
    let daysUntilExpiry: Int
    let isExpired: Bool
    
    var daysUntilExpiryText: String {
        if isExpired {
            return "Expired"
        } else if daysUntilExpiry == 0 {
            return "Today"
        } else if daysUntilExpiry == 1 {
            return "1 day"
        } else {
            return "\(daysUntilExpiry) days"
        }
    }
    
    var expiryColor: Color {
        if isExpired {
            return .red
        } else if daysUntilExpiry <= 7 {
            return .orange
        } else {
            return .yellow
        }
    }
}
```

```swift
import Foundation
import CoreData

class WidgetDataManager {
    static let shared = WidgetDataManager()
    private let suiteName = "group.com.yourcompany.medicinetime"
    
    private var sharedDefaults: UserDefaults? {
        UserDefaults(suiteName: suiteName)
    }
    
    func updateWidgetData(medications: [Medication]) {
        let totalCount = medications.count
        let expiringCount = medications.filter { $0.expiryStatus == .expiringSoon || $0.expiryStatus == .expired }.count
        let lowStockCount = medications.filter { $0.needsRestock }.count
        
        // Update counts
        sharedDefaults?.set(totalCount, forKey: "widget_totalCount")
        sharedDefaults?.set(expiringCount, forKey: "widget_expiringCount")
        sharedDefaults?.set(lowStockCount, forKey: "widget_lowStockCount")
        
        // Update expiring medications list
        let expiringMeds = medications
            .filter { $0.expiryStatus == .expiringSoon || $0.expiryStatus == .expired }
            .sorted { $0.daysUntilExpiry < $1.daysUntilExpiry }
            .map { med -> WidgetMedication in
                WidgetMedication(
                    name: med.displayName,
                    daysUntilExpiry: med.daysUntilExpiry,
                    isExpired: med.isExpired
                )
            }
        
        if let encoded = try? JSONEncoder().encode(expiringMeds) {
            sharedDefaults?.set(encoded, forKey: "widget_expiringMedications")
        }
        
        // Trigger widget refresh
        WidgetCenter.shared.reloadAllTimelines()
    }
}
```

```swift
// MARK: - Live Activity Attributes
struct MedicationReminderAttributes: ActivityAttributes {
    public struct ContentState: Codable, Hashable {
        var medicationName: String
        var quantity: Int16
        var unit: String
        var reminderTime: Date
        var isOverdue: Bool
    }
    
    var medicationID: String
    var medicationName: String
}

// MARK: - Live Activity Manager
@available(iOS 16.1, *)
class LiveActivityManager {
    static let shared = LiveActivityManager()
    
    private var currentActivity: Activity<MedicationReminderAttributes>?
    
    func startMedicationReminder(medication: Medication) {
        guard ActivityAuthorizationInfo().areActivitiesEnabled else {
            print("Live Activities not enabled")
            return
        }
        
        let attributes = MedicationReminderAttributes(
            medicationID: medication.id?.uuidString ?? UUID().uuidString,
            medicationName: medication.displayName
        )
        
        let initialState = MedicationReminderAttributes.ContentState(
            medicationName: medication.displayName,
            quantity: medication.quantity,
            unit: medication.displayUnit,
            reminderTime: Date(),
            isOverdue: false
        )
        
        do {
            let activity = try Activity.request(
                attributes: attributes,
                contentState: initialState,
                pushType: nil
            )
            currentActivity = activity
            print("Started Live Activity: \(activity.id)")
        } catch {
            print("Error starting Live Activity: \(error)")
        }
    }
    
    func updateMedicationReminder(quantity: Int16, isOverdue: Bool = false) {
        guard let activity = currentActivity else { return }
        
        let updatedState = MedicationReminderAttributes.ContentState(
            medicationName: activity.attributes.medicationName,
            quantity: quantity,
            unit: "tablets", // Get from medication
            reminderTime: activity.contentState.reminderTime,
            isOverdue: isOverdue
        )
        
        Task {
            await activity.update(using: updatedState)
        }
    }
    
    func endMedicationReminder() {
        guard let activity = currentActivity else { return }
        
        Task {
            await activity.end(dismissalPolicy: .default)
            currentActivity = nil
        }
    }
}
```

### 3.1 P0: 处方药标记功能 (Prescription Flag)

#### 问题描述
美国用户需要区分处方药(Rx)和非处方药(OTC)，处方药需要特殊管理和续方跟踪。

#### 需求描述
添加处方药标记功能：
1. **处方药标记** - 在添加/编辑药物时标记是否为处方药
2. **Rx图标显示** - 在药物卡片和详情页显示Rx标识
3. **续方日期跟踪** - 记录处方续方日期(Refill Date)

#### 数据模型修改

**修改 - Medication+Extensions.swift:**
```swift
// MARK: - Prescription Properties
extension Medication {
    public var isPrescription: Bool {
        get { isPrescription_ }
        set { isPrescription_ = newValue }
    }
    
    public var refillDate: Date? {
        get { refillDate_ }
        set { refillDate_ = newValue }
    }
    
    public var prescriptionNumber: String? {
        get { prescriptionNumber_ }
        set { prescriptionNumber_ = newValue }
    }
    
    public var displayPrescriptionInfo: String {
        if isPrescription {
            if let refill = refillDate {
                return "Rx • Refill: \(formatDate(refill))"
            }
            return "Rx Only"
        }
        return "OTC"
    }
    
    public var needsRefill: Bool {
        guard isPrescription, let refill = refillDate else { return false }
        return refill <= Date().addingTimeInterval(7 * 86400) // Within 7 days
    }
    
    private func formatDate(_ date: Date) -> String {
        let formatter = DateFormatter()
        formatter.dateStyle = .short
        return formatter.string(from: date)
    }
}
```

### 3.3 P2: FDA药物数据库链接

#### 问题描述

#### 需求描述
添加FDA药物数据库链接：
1. **FDA查询按钮** - 在药物详情页添加查询按钮
2. **自动搜索** - 使用药物名称自动构建FDA查询URL

#### 生成规则

**代码编写 - FDAService.swift:**
```swift
//
//  FDAService.swift
//  medicinetime
//

import Foundation

class FDAService {
    static let shared = FDAService()
    
    private let fdaBaseURL = "https://www.accessdata.fda.gov/scripts/cder/daf/"
    
    func openFDADrugInfo(for medicationName: String) {
        let encodedName = medicationName.addingPercentEncoding(withAllowedCharacters: .urlQueryAllowed) ?? medicationName
        let urlString = "\(fdaBaseURL)?event=overview.process&varApplNo=&varProductNo=&varTradeName=\(encodedName)"
        
        if let url = URL(string: urlString) {
            UIApplication.shared.open(url)
        }
    }
    
    func openFDADrugDatabase() {
        if let url = URL(string: "https://www.accessdata.fda.gov/scripts/cder/daf/") {
            UIApplication.shared.open(url)
        }
    }
}
```

   | pharmacyPhone_ | String | YES | 药店电话 |

   | insurancePolicyNumber_ | String | YES | 保单号 |
   | copayAmount_ | Double | NO | 共付额(默认0) |

   - 在PersistenceController中启用轻量级迁移

#### 代码修改 - PersistenceController.swift

```swift
// MARK: - Core Data Stack with Migration Support
lazy var container: NSPersistentContainer = {
    let container = NSPersistentContainer(name: "MedicationModel")
    
    // Enable lightweight migration
    let description = NSPersistentStoreDescription()
    description.shouldMigrateStoreAutomatically = true
    description.shouldInferMappingModelAutomatically = true
    container.persistentStoreDescriptions = [description]
    
    container.loadPersistentStores { description, error in
        if let error = error {
            fatalError("Core Data Store failed: \(error.localizedDescription)")
        }
    }
    
    container.viewContext.automaticallyMergesChangesFromParent = true
    container.viewContext.mergePolicy = NSMergeByPropertyObjectTrumpMergePolicy
    
    return container
}()
```
